using System;
using System.IO;
using System.Xml;

public class Sample
{
  public static void Main()
  {
     XmlDocument doc = new XmlDocument();
     doc.LoadXml("<book genre='novel' publicationdate='1997'> " +
                       "  <title>Pride And Prejudice</title>" +
                       "</book>");

     for (int i=0; i < attrColl.Count; i++)

     }         
    
  }

using System;
using System.IO;
using System.Xml;

public class Sample
{
  public static void Main(){
  
    XmlDocument doc = new XmlDocument();

                "<title>Pride And Prejudice</title>" +

    } 
  
  }
